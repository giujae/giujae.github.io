    if max(numbers) == 1:
        return 0
    elif max(numbers) == 2:
        return 1
    elif max(numbers) == 3:
        if 2 in numbers:
            return 2
        else:
            return 1

            else:
                prime.append(num)
    for number in numbers:
        if number in prime:
            cnt += 1
    return cnt

    prime = [False, False] + [True] * (n - 1)
    for i in range(4, n+1, 2):
        prime[i] = False
    for p in range(3, int(n ** 0.5) + 1, 2):

            for j in range(p*p, n+1, p):
                prime[j] = False
    return [p for p in range(2, n+1) if prime[p]]
